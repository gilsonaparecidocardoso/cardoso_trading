#GIT
#Comandos Git básicos para o fluxo de trabalho
#Para usar o GIT no console Incluir no arquivo C:\Users\user\AppData\Roaming\Code\User\settings.json
"terminal.integrated.defaultProfile.windows": "Git Bash",
"terminal.integrated.tabs.enabled": true
#ATALHO CTRL+SHIFT+G

#Com a pasta raiz aberta cria o repositorio
git init
git add .                                   : Adiciona todos os arquivos modificados para a área de preparação (staging).
git commit -m "sua mensagem"                : Cria um novo commit com as alterações e uma mensagem descritiva.
git push                                    : Envia seus commits locais para o repositório remoto (como no GitHub).
git remote add origin [URL do repositório]  : Conecta o seu repositório local a um repositório remoto.
git clone [URL do repositório]              : Clona um repositório remoto existente para o seu computador. 

#…or create a new repository on the command line
echo "# cardoso_trading" >> README.md
git init
git add README.md
git commit -m "first commit"
git branch -M main
git remote add origin https://github.com/gilsonaparecidocardoso/cardoso_trading.git
git push -u origin main

#…or push an existing repository from the command line
git remote add origin https://github.com/gilsonaparecidocardoso/cardoso_trading.git
git branch -M main
git push -u origin main
